    <script>

        // --- App State ---

        const DAILY_BUDGET = 50;

        let currentDate = new Date();

        let viewingMonth = new Date();

        let selectedDate = new Date(); // Tracks which date we're viewing/editing

        let expenses = loadExpenses(); // Load saved data on start

        

        // --- Data Persistence ---

        

        /**

         * Saves the expenses object to browser cookies.

         * @param {object} expenses - The expenses object to save.

         */

        function saveExpenses(expenses) {

            try {

                const data = JSON.stringify(expenses);

                // Set a cookie that expires in the future

                document.cookie = `budgetExpenses=${encodeURIComponent(data)}; expires=Fri, 31 Dec 2025 23:59:59 GMT; path=/`;

            } catch (e) {

                console.error('Could not save expenses:', e);

            }

        }


        /**

         * Loads expenses from browser cookies.

         * @returns {object} The loaded expenses object or an empty object.

         */

        function loadExpenses() {

            try {

                const cookies = document.cookie.split(';');

                for (let cookie of cookies) {

                    const [name, value] = cookie.trim().split('=');

                    if (name === 'budgetExpenses' && value) {

                        return JSON.parse(decodeURIComponent(value));

                    }

                }

            } catch (e) {

                console.error('Could not load expenses:', e);

            }

            return {}; // Return empty object if nothing is found

        }

        

        /**

         * Ensures every expense has a unique ID for reliable deletion.

         * This is for backward compatibility if old data exists without IDs.

         */

        function ensureExpenseIds() {

            let needsSave = false;

            for (const dateKey in expenses) {

                expenses[dateKey] = expenses[dateKey].map(expense => {

                    if (!expense.id) {

                        expense.id = Date.now() + Math.random();

                        needsSave = true;

                    }

                    return expense;

                });

            }

            if (needsSave) {

                saveExpenses(expenses);

            }

        }

        

        // --- Core Logic ---


        /**

         * Creates a standardized key from a date object (e.g., "2023-10-27").

         * @param {Date} date - The date to convert.

         * @returns {string} The date key.

         */

        function getDateKey(date) {

            return date.toISOString().split('T')[0];

        }


        /**

         * Retrieves expenses for a specific date.

         * @param {Date} date - The date to get expenses for.

         * @returns {Array} An array of expense objects.

         */

        function getExpensesForDate(date) {

            const dateKey = getDateKey(date);

            return expenses[dateKey] || [];

        }


        /**

         * Adds a new expense to the selected date.

         * @param {string} description - The expense description.

         * @param {number} amount - The expense amount.

         */

        function addExpense(description, amount) {

            const targetDateKey = getDateKey(selectedDate);

            if (!expenses[targetDateKey]) {

                expenses[targetDateKey] = [];

            }

            

            const newExpense = {

                id: Date.now() + Math.random(), // Unique ID

                description,

                amount: parseFloat(amount),

                timestamp: new Date().toISOString()

            };

            

            expenses[targetDateKey].push(newExpense);

            saveExpenses(expenses);

            updateUI(); // Refresh the entire display

        }


        /**

         * Deletes an expense by its ID and date key.

         * @param {string} dateKey - The key for the date of the expense.

         * @param {number} expenseId - The unique ID of the expense to delete.

         */

        function deleteExpense(dateKey, expenseId) {

            if (expenses[dateKey]) {

                expenses[dateKey] = expenses[dateKey].filter(exp => exp.id !== expenseId);

                // If a date has no more expenses, remove it from the object

                if (expenses[dateKey].length === 0) {

                    delete expenses[dateKey];

                }

                saveExpenses(expenses);

                updateUI(); // Refresh the entire display

            }

        }


        // --- UI Updates ---


        /**

         * Main function to update all parts of the UI.

         */

        function updateUI() {

            updateDailyBudgetHeader();

            updateExpensesList();

            updateMonthlySummary();

            updateCalendar();

        }


        /**

         * Updates the header with the budget status for the selected date.

         */

        function updateDailyBudgetHeader() {

            const selectedExpenses = getExpensesForDate(selectedDate);

            const totalSpent = selectedExpenses.reduce((sum, exp) => sum + exp.amount, 0);

            const remaining = DAILY_BUDGET - totalSpent;

            const statusElement = document.getElementById('budget-status');

            

            document.getElementById('remaining-budget').textContent = Math.abs(remaining).toFixed(2);

            

            const isToday = getDateKey(selectedDate) === getDateKey(currentDate);

            const dateText = isToday ? 'today' : `on ${selectedDate.toLocaleDateString('en-US', { month: 'short', day: 'numeric' })}`;

            

            if (remaining > 0) {

                statusElement.textContent = `$${remaining.toFixed(2)} remaining ${dateText}`;

                statusElement.className = 'budget-status under-budget';

            } else if (remaining < 0) {

                statusElement.textContent = `$${Math.abs(remaining).toFixed(2)} over budget ${dateText}`;

                statusElement.className = 'budget-status over-budget';

            } else {

                statusElement.textContent = `Exactly on budget ${dateText}!`;

                statusElement.className = 'budget-status on-budget';

            }

        }


        /**

         * Updates the list of expenses for the selected date.

         */

        function updateExpensesList() {

            const selectedExpenses = getExpensesForDate(selectedDate);

            const container = document.getElementById('expenses-container');

            const dateLabel = document.getElementById('expenses-date');

            

            // Update the title (e.g., "Today's Expenses" or "Oct 27's Expenses")

            if (getDateKey(selectedDate) === getDateKey(currentDate)) {

                dateLabel.textContent = "Today's";

            } else {

                dateLabel.textContent = selectedDate.toLocaleDateString('en-US', { month: 'short', day: 'numeric' }) + "'s";

            }

            

            container.innerHTML = ''; // Clear previous list

            

            if (selectedExpenses.length === 0) {

                const dateText = getDateKey(selectedDate) === getDateKey(currentDate) ? 'today' : 'this day';

                container.innerHTML = `<p style="text-align: center; color: #bbb; font-style: italic;">No expenses yet ${dateText}</p>`;

            } else {

                selectedExpenses.forEach(expense => {

                    const expenseDiv = document.createElement('div');

                    expenseDiv.className = 'expense-item';

                    

                    expenseDiv.innerHTML = `

                        <div class="expense-content">

                            <span class="expense-description">${expense.description}</span>

                            <span class="expense-amount">$${expense.amount.toFixed(2)}</span>

                        </div>

                        <button class="delete-btn" title="Delete expense">×</button>

                    `;

                    

                    // Add event listener to the delete button

                    expenseDiv.querySelector('.delete-btn').addEventListener('click', () => {

                        deleteExpense(getDateKey(selectedDate), expense.id);

                    });

                    

                    container.appendChild(expenseDiv);

                });

            }

        }


        /**

         * Updates the monthly summary based on the current month.

         */

        function updateMonthlySummary() {

            const month = currentDate.getMonth();

            const year = currentDate.getFullYear();

            const daysInMonth = new Date(year, month + 1, 0).getDate();

            const monthlyBudget = DAILY_BUDGET * daysInMonth;


            let monthlySpent = 0;

            for (const dateKey in expenses) {

                const expenseDate = new Date(dateKey); // Add time to avoid timezone issues

                if (expenseDate.getMonth() === month && expenseDate.getFullYear() === year) {

                    monthlySpent += expenses[dateKey].reduce((sum, exp) => sum + exp.amount, 0);

                }

            }


            document.getElementById('monthly-budget').textContent = monthlyBudget.toFixed(2);

            document.getElementById('monthly-spent').textContent = monthlySpent.toFixed(2);

            document.getElementById('monthly-remaining').textContent = (monthlyBudget - monthlySpent).toFixed(2);

        }


        /**

         * Renders the calendar for the currently viewed month.

         */

        function updateCalendar() {

            const grid = document.getElementById('calendar-grid');

            const monthYearEl = document.getElementById('current-month-year');

            

            const year = viewingMonth.getFullYear();

            const month = viewingMonth.getMonth();

            

            monthYearEl.textContent = viewingMonth.toLocaleDateString('en-US', { month: 'long', year: 'numeric' });

            grid.innerHTML = '';


            // Add day headers (S, M, T, W, T, F, S)

            ['S', 'M', 'T', 'W', 'T', 'F', 'S'].forEach(day => {

                const dayEl = document.createElement('div');

                dayEl.className = 'calendar-day header';

                dayEl.textContent = day;

                grid.appendChild(dayEl);

            });


            // Calculate the start date for the calendar grid

            const firstDayOfMonth = new Date(year, month, 1);

            const startDate = new Date(firstDayOfMonth);

            startDate.setDate(startDate.getDate() - firstDayOfMonth.getDay());


            // Create 42 days for the grid (6 weeks)

            for (let i = 0; i < 42; i++) {

                const date = new Date(startDate);

                date.setDate(date.getDate() + i);

                

                const dayEl = document.createElement('div');

                dayEl.className = 'calendar-day';

                dayEl.textContent = date.getDate();


                const isCurrentViewingMonth = date.getMonth() === month;

                

                if (isCurrentViewingMonth) {

                    dayEl.classList.add('current-month', 'clickable');

                    

                    // Style the day based on spending

                    const daySpent = getExpensesForDate(date).reduce((sum, exp) => sum + exp.amount, 0);

                    if (daySpent > 0) {

                        if (daySpent < DAILY_BUDGET) dayEl.classList.add('under-budget');

                        else if (daySpent > DAILY_BUDGET) dayEl.classList.add('over-budget');

                        else dayEl.classList.add('on-budget');

                    }

                    

                    // Highlight today and the selected date

                    if (date.toDateString() === currentDate.toDateString()) dayEl.classList.add('today');

                    if (date.toDateString() === selectedDate.toDateString()) dayEl.classList.add('selected');

                    

                    // Add click handler to view expenses for this date

                    dayEl.addEventListener('click', () => {

                        selectedDate = date;

                        updateUI();

                    });

                } else {

                    dayEl.classList.add('other-month');

                }


                grid.appendChild(dayEl);

            }

        }


        // --- Event Listeners ---

        

        // Handle form submission to add a new expense

        document.getElementById('expense-form').addEventListener('submit', (e) => {

            e.preventDefault();

            const descriptionInput = document.getElementById('expense-description');

            const amountInput = document.getElementById('expense-amount');

            

            if (descriptionInput.value && amountInput.value > 0) {

                addExpense(descriptionInput.value, amountInput.value);

                e.target.reset(); // Clear the form

                descriptionInput.focus();

            }

        });


        // Navigate to the previous month in the calendar

        document.getElementById('prev-month').addEventListener('click', () => {

            viewingMonth.setMonth(viewingMonth.getMonth() - 1);

            updateCalendar();

        });


        // Navigate to the next month in the calendar

        document.getElementById('next-month').addEventListener('click', () => {

            viewingMonth.setMonth(viewingMonth.getMonth() + 1);

            updateCalendar();

        });

        // Run on page load

        function initialize() {

            ensureExpenseIds();

            updateUI();

        }


        initialize();
